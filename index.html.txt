  <div class="card">
    <h2>üåç Currency Converter (Offline - 120+)</h2>
    <input type="number" id="amount" placeholder="Enter Amount" />
    <br>
    <select id="from"></select>
    <select id="to"></select>
    <br>
    <button onclick="convert()">Convert</button>
    <h3 id="result">Converted Amount: </h3>
  </div>

  <script>
    let rates = {};

    // Load rates from JSON file
    fetch("currencies.json")
      .then(response => response.json())
      .then(data => {
        rates = data;

        const fromSelect = document.getElementById("from");
        const toSelect = document.getElementById("to");

        Object.keys(rates).forEach(currency => {
          fromSelect.innerHTML += `<option value="${currency}">${currency}</option>`;
          toSelect.innerHTML += `<option value="${currency}">${currency}</option>`;
        });

        // Default: INR to USD
        fromSelect.value = "INR";
        toSelect.value = "USD";
      });

    function convert() {
      const amount = parseFloat(document.getElementById("amount").value);
      const from = document.getElementById("from").value;
      const to = document.getElementById("to").value;

      if (isNaN(amount)) {
        alert("‚ö†Ô∏è Please enter a valid number");
        return;
      }

      // Convert via INR (base currency)
      const inINR = amount * rates[from];
      const converted = inINR / rates[to];

      document.getElementById("result").innerText =
        `Converted Amount: ${converted.toFixed(2)} ${to}`;
    }
  </script>
